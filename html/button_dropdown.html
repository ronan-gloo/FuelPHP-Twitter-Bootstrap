// nested list
$nest = Html::dropdown();
$nest->item('#', 'messages', 'inbox');
$nest->item('#', 'logout', 'off');

// dropdown
$btn = Html::button_dropdown('#', "Don't Do This !", array(
	'split' => true,
	'status'=> 'warning',
	'icon'	=> 'warning-sign'
));
$btn->item('#', 'Home', 'home');
$btn->item('#', 'Pages', 'file');
$btn->item('#', 'Account', 'cog')->nest($nest);
